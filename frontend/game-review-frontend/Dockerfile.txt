# Etapa 1: build
FROM node:18-alpine AS builder

WORKDIR /app

# Copia apenas arquivos de dependências primeiro
COPY package*.json ./

# Instala todas as dependências (inclui devDependencies como o Vite)
RUN npm install

# Copia o resto do código
COPY . .

# Faz o build do projeto com Vite
RUN npm run build


# Etapa 2: runtime (produção)
FROM node:18-alpine

WORKDIR /app

# Copia apenas os arquivos de build da etapa anterior
COPY --from=builder /app/dist ./dist
COPY package*.json ./

# Instala apenas as dependências necessárias em produção
RUN npm install --omit=dev

# Expor a porta usada pelo Vite preview
EXPOSE 4173

# Comando para rodar o preview do build
CMD ["npx", "vite", "preview", "--host", "0.0.0.0", "--port", "4173"]
