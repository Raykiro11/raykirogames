# Etapa 1: build
FROM node:18-alpine AS builder
WORKDIR /app

COPY package*.json ./
# usa npm ci para instalar todas as dependências (inclusive dev)
RUN npm ci

COPY . .
RUN npm run build

# Etapa 2: runtime (apenas o que é necessário para rodar)
FROM node:18-alpine
WORKDIR /app

COPY --from=builder /app/dist ./dist
COPY package*.json ./

# instala apenas dependências de produção
RUN npm ci --omit=dev

EXPOSE 4173
CMD ["npx", "vite", "preview", "--host", "0.0.0.0", "--port", "4173"]
