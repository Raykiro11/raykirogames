FROM node:18-alpine

WORKDIR /app

# Copia apenas os arquivos de dependências
COPY package*.json ./

# Instala todas as dependências (incluindo devDependencies)
RUN npm install

# Copia o restante do projeto
COPY . .

# Build do Vite
RUN npm run build

# Porta padrão do Vite preview
EXPOSE 4173

# Comando para iniciar o app
CMD ["npm", "run", "preview"]
