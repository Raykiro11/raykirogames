# Usar Node 18 como base
FROM node:18-alpine

# Definir diretório de trabalho
WORKDIR /app

# Copiar apenas os arquivos de dependências primeiro
COPY package*.json ./

# Instalar TODAS as dependências (não omite dev)
RUN npm install

# Copiar o resto do código
COPY . .

# Fazer build do Vite
RUN npm run build

# Expor a porta do preview
EXPOSE 4173

# Rodar o app com preview
CMD ["npm", "run", "preview"]
