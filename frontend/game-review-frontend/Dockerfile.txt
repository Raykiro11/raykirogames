# Etapa 1: build
FROM node:18-alpine AS builder

WORKDIR /app

# Copiar apenas arquivos de dependências
COPY package*.json ./

# Instalar TODAS as dependências (inclui dev)
RUN npm install

# Copiar todo o código
COPY . .

# Build do Vite
RUN npm run build

# Etapa 2: runtime (produção)
FROM node:18-alpine

WORKDIR /app

# Copiar o build da etapa anterior
COPY --from=builder /app/dist ./dist
COPY package*.json ./

# Instalar apenas dependências de produção
RUN npm install --omit=dev

# Expor porta do Vite preview
EXPOSE 4173

# Rodar o frontend
CMD ["npx", "vite", "preview", "--host", "0.0.0.0", "--port", "4173"]
