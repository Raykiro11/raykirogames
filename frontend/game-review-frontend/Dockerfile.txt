# Etapa 1: build (tem que ter o Vite aqui)
FROM node:18-alpine AS builder
WORKDIR /app

# Copia configs e instala TODAS as deps (incluindo dev)
COPY package*.json ./
RUN npm install

# Copia o resto e roda build
COPY . .
RUN npm run build

# Etapa 2: runtime (mais leve, sem devDeps)
FROM node:18-alpine
WORKDIR /app

# Copia somente o dist gerado e package.json
COPY --from=builder /app/dist ./dist
COPY package*.json ./

# Instala apenas deps de produção
RUN npm install --omit=dev

EXPOSE 4173
CMD ["npx", "vite", "preview", "--host", "0.0.0.0", "--port", "4173"]
