# Etapa 1: build
FROM node:18-alpine AS builder
WORKDIR /app

# Copia configs e instala dependências (com devDeps, para o Vite estar disponível)
COPY package*.json ./
RUN npm install

# Copia o resto e roda build
COPY . .
RUN npm run build

# Etapa 2: runtime
FROM node:18-alpine
WORKDIR /app

# Copia apenas o necessário para rodar
COPY --from=builder /app/dist ./dist
COPY package*.json ./

# Instala apenas as deps de produção (sem dev)
RUN npm install --omit=dev

EXPOSE 4173
CMD ["npx", "vite", "preview", "--host", "0.0.0.0", "--port", "4173"]
